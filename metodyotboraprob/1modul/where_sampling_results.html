var state = {

  opened: { pec:false, nvoc:false, plan:false, complaint:false, incident:false },

  currentKey: null,

  currentPart: 0

};


var cards = document.querySelectorAll('.card');

var modal = document.getElementById('modal');

var modalTitle = document.getElementById('modal-title');

var modalChip = document.getElementById('modal-chip');

var modalBody = document.getElementById('modal-body');

var modalClose = document.getElementById('modal-close');

var modalPrev = document.getElementById('modal-prev');

var modalNext = document.getElementById('modal-next');

var progressLabel = document.getElementById('progress-label');

var progressBarInner = document.getElementById('progress-bar-inner');

var finalNote = document.getElementById('final-note');


cards.forEach(function(card){

  card.addEventListener('click', function() {

    var key = card.getAttribute('data-key');

    openScenario(key);

  });

});


modalClose.addEventListener('click', function() {

  modal.style.display = 'none';

});


modalPrev.addEventListener('click', function() {

  if (!state.currentKey) return;

  if (state.currentPart > 0) {

    state.currentPart -= 1;

    renderModal();

  }

});


modalNext.addEventListener('click', function() {

  if (!state.currentKey) return;

  var parts = scenarios[state.currentKey].parts;

  if (state.currentPart < parts.length - 1) {

    state.currentPart += 1;

    renderModal();

  } else {

    // конец сценария

    state.opened[state.currentKey] = true;

    updateCards();

    updateProgress();

    modal.style.display = 'none';

  }

});


function openScenario(key) {

  state.currentKey = key;

  state.currentPart = 0;

  renderModal();

  modal.style.display = 'flex';

}


function renderModal() {

  var sc = scenarios[state.currentKey];

  var part = sc.parts[state.currentPart];


  modalTitle.textContent = sc.title;

  modalChip.textContent = sc.chip;


  modalBody.innerHTML = '';

  var p1 = document.createElement('p');

  p1.textContent = part.text;

  modalBody.appendChild(p1);


  if (part.hint) {

    var div = document.createElement('div');

    div.className = 'highlight';

    div.textContent = part.hint;

    modalBody.appendChild(div);

  }


  // кнопки назад/дальше

  if (state.currentPart === 0) {

    modalPrev.disabled = true;

    modalPrev.style.opacity = '0.5';

  } else {

    modalPrev.disabled = false;

    modalPrev.style.opacity = '1';

  }


  if (state.currentPart === sc.parts.length - 1) {

    modalNext.textContent = 'Закрыть ситуацию';

  } else {

    modalNext.textContent = 'Дальше';

  }

}


function updateCards() {

  cards.forEach(function(card) {

    var key = card.getAttribute('data-key');

    var status = card.querySelector('.card-status');

    if (state.opened[key]) {

      card.classList.add('done');

      status.textContent = 'Просмотрено';

    } else {

      card.classList.remove('done');

      status.textContent = 'Не изучено';

    }

  });

}


function updateProgress() {

  var keys = Object.keys(state.opened);

  var done = keys.filter(function(k){ return state.opened[k]; }).length;

  var total = keys.length;

  progressLabel.textContent = 'Открыто ситуаций: ' + done + ' из ' + total;

  var percent = (done / total) * 100;

  progressBarInner.style.width = percent + '%';


  if (done === total) {

    finalNote.style.display = 'block';

  }

}


// инициализация

updateCards();

updateProgress();
